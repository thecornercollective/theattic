        // search games with popup
        function searchgames() {
            const searchterm = document.getElementById('searchInput').value.toLowerCase().trim();
            const searchresults = document.getElementById('searchResults');

            if (searchterm === '') {
                searchresults.classList.remove('active');
                searchresults.innerHTML = '';
                return;
            }

            const filteredgames = games.filter(game => 
                game.name.toLowerCase().includes(searchterm)
            );

            if (filteredgames.length === 0) {
                searchresults.innerHTML = '<div class="no-results">no games found</div>';
            } else {
                searchresults.innerHTML = filteredgames.map(game => `
                    <a href="play.html?game=${game.id}" class="search-result-item">
                        <img src="${game.img}" alt="${game.name}">
                        <div class="info">
                            <div class="name">${game.name}</div>
                            <div class="category">${game.category}</div>
                        </div>
                    </a>
                `).join('');
            }

            searchresults.classList.add('active');

        // Initialize Scramjet
        async function initScramjet() {
            updateProxyStatus('loading', 'Initializing proxy...');
            
            try {
                // Load Scramjet Controller
                const { ScramjetController } = $scramjetLoadController();
                scramjetController = new ScramjetController({
                    files: {
                        wasm: "https://cdn.jsdelivr.net/npm/@mercuryworkshop/scramjet@1.2.2/scramjet.wasm.wasm",
                        all: "https://cdn.jsdelivr.net/npm/@mercuryworkshop/scramjet@1.2.2/scramjet.all.js",
                        sync: "https://cdn.jsdelivr.net/npm/@mercuryworkshop/scramjet@1.2.2/scramjet.sync.js",
                    },
                    flags: {
                        rewriterLogs: false,
                        scramitize: false,
                        cleanErrors: true,
                        sourcemaps: true,
                    },
                });

                // Initialize Scramjet
                await scramjetController.init();
                updateProxyStatus('loading', 'Registering service worker...');
                
                // Register service worker
                if ('serviceWorker' in navigator) {
                    try {
                        await navigator.serviceWorker.register('./sw.js');
                        console.log('Service Worker registered successfully');
                    } catch (swError) {
                        console.warn('Service Worker registration failed:', swError);
                        updateProxyStatus('offline', 'Service worker failed');
                    }
                }

                // Setup transport
                try {
                    updateProxyStatus('loading', 'Configuring transport...');
                    const connection = new BareMux.BareMuxConnection("https://cdn.jsdelivr.net/npm/@mercuryworkshop/bare-mux@0.1.2/dist/worker.js");
                    await connection.setTransport(
                        "https://cdn.jsdelivr.net/npm/@mercuryworkshop/epoxy-transport@0.0.15/dist/index.mjs",
                        [{ wisp: "wss://wisp.mercurywork.shop/" }]
                    );
                    console.log('Transport configured successfully');
                } catch (transportError) {
                    console.warn('Transport setup failed:', transportError);
                    updateProxyStatus('offline', 'Transport failed');
                }

                console.log('Scramjet initialized successfully');
                updateProxyStatus('online', 'Proxy connected');
                
                // Auto-hide status after 3 seconds if successful
                setTimeout(() => {
                    const statusEl = document.getElementById('proxyStatus');
                    if (statusEl && statusEl.classList.contains('online')) {
                        statusEl.style.opacity = '0';
                        setTimeout(() => {
                            if (statusEl) {
                                statusEl.style.display = 'none';
                            }
                        }, 300);
                    }
                }, 3000);
                
                return scramjetController;
            } catch (error) {
                console.error('Scramjet initialization failed:', error);
                updateProxyStatus('offline', 'Proxy failed');
                return null;
            }

        // Close tab
        function closeTab(tabId) {
            const tabIndex = tabs.findIndex(t => t.id === tabId);
            if (tabIndex === -1) return;
            
            // Clean up frame if exists
            const tab = tabs[tabIndex];
            if (tab.frame && tab.frame.frame) {
                tab.frame.frame.remove();
            }
            
            tabs.splice(tabIndex, 1);
            
            if (tabs.length === 0) {
                createNewTab();
            } else if (activeTabId === tabId) {
                // Switch to the last tab if the active tab was closed
                const newIndex = Math.min(tabIndex, tabs.length - 1);
                activeTabId = tabs[newIndex].id;
                renderTabs();
                updateBrowserContent();
            } else {
                renderTabs();
            }
        }

        // Navigate to URL
        function navigateTo(url) {
            const tab = tabs.find(t => t.id === activeTabId);
            if (!tab) return;
            
            // Ensure url is a string
            if (typeof url !== 'string') {
                url = String(url);
            }
            
            // Get safe URL
            url = getSafeUrl(url);
            
            // Validate URL
            if (!isValidUrl(url)) {
                showError('Invalid URL. Please enter a valid web address.');
                return;
            }
            
            // Update tab data
            tab.url = url;
            tab.title = getDomainName(url);
            
            // Add to history
            if (tab.historyIndex < tab.history.length - 1) {
                tab.history = tab.history.slice(0, tab.historyIndex + 1);
            }
            tab.history.push(url);
            tab.historyIndex = tab.history.length - 1;
            
            renderTabs();
            updateBrowserContent();
            showLoading();
        }

        // Get domain name from URL
        function getDomainName(url) {
            try {
                if (!url || typeof url !== 'string') {
                    return 'Page';
                }
                const urlObj = new URL(url);
                return urlObj.hostname.replace('www.', '');
            } catch {
                return 'Page';
            }
        }

        // Update safe URL function to use custom search engine
        function getSafeUrl(input) {
            let url = input.trim();
            
            // Check if it's a search query
            if (!url.includes('.') && !url.startsWith('http')) {
                return searchEngines[getSearchEngine()] + encodeURIComponent(url);
            }
            
            // Add protocol if missing
            if (!url.startsWith('http://') && !url.startsWith('https://')) {
                url = 'https://' + url;
            }
            
            return url;

        function encodeUrl(url) {
            try {
                return encodeURIComponent(url);
            } catch (e) {
                return url;

        }

        // Handle URL decoding
        function decodeUrl(encodedUrl) {
            try {
                return decodeURIComponent(encodedUrl);
            } catch (e) {
                return encodedUrl;

        // Get URL parameters
        function getUrlParameters(url) {
            try {
                const urlObj = new URL(url);
                const params = {};
                urlObj.searchParams.forEach((value, key) => {

                return params;
            } catch (e) {
                return {};

        // Export tabs data
        function exportTabs() {
            const tabsData = tabs.map(tab => ({
                id: tab.id,
                title: tab.title,
                url: tab.url,
                history: tab.history
            }));
            
            const dataStr = JSON.stringify(tabsData, null, 2);
            const dataBlob = new Blob([dataStr], { type: 'application/json' });
            const url = URL.createObjectURL(dataBlob);
            
            const link = document.createElement('a');
            link.href = url;
            link.download = 'veinte-tabs.json';
            link.click();
            
            URL.revokeObjectURL(url);
        }

        // Import tabs data
        function importTabs(file) {
            const reader = new FileReader();
            reader.onload = (e) => {
                try {
                    const tabsData = JSON.parse(e.target.result);
                    tabsData.forEach(tabData => {
                        const tab = {
                            id: ++tabCounter,
                            title: tabData.title || 'Imported Tab',
                            url: tabData.url || null,
                            history: tabData.history || [],
                            historyIndex: tabData.history ? tabData.history.length - 1 : -1,
                            frame: null
                        };
                        tabs.push(tab);
                    });
                    renderTabs();
                    console.log('Tabs imported successfully');
                } catch (err) {
                    console.error('Failed to import tabs:', err);
                }
            };
            reader.readAsText(file);
        }

        function getTimeSince(date) {
            const now = new Date();
            const past = new Date(date);
            const diff = now - past;
            
            const seconds = Math.floor(diff / 1000);
            const minutes = Math.floor(seconds / 60);
            const hours = Math.floor(minutes / 60);
            const days = Math.floor(hours / 24);
            
            if (days > 0) return `${days} day${days > 1 ? 's' : ''} ago`;
            if (hours > 0) return `${hours} hour${hours > 1 ? 's' : ''} ago`;
            if (minutes > 0) return `${minutes} minute${minutes > 1 ? 's' : ''} ago`;
            return 'Just now';

        // Go back
        function goBack() {
            const tab = tabs.find(t => t.id === activeTabId);
            if (!tab || tab.historyIndex <= 0) return;
            
            tab.historyIndex--;
            tab.url = tab.history[tab.historyIndex];
            tab.title = getDomainName(tab.url);
            
            renderTabs();
            updateBrowserContent();
            showLoading();

        // Go forward

            const tab = tabs.find(t => t.id === activeTabId);
            if (!tab || tab.historyIndex >= tab.history.length - 1) return;
            
            tab.historyIndex++;
            tab.url = tab.history[tab.historyIndex];
            tab.title = getDomainName(tab.url);
            
            renderTabs();
            updateBrowserContent();

        // Keyboard shortcuts
        document.addEventListener('keydown', (e) => {
            // Check if user is typing in input fields
            const activeElement = document.activeElement;
            const isInputFocused = activeElement.tagName === 'INPUT' || 
                                  activeElement.tagName === 'TEXTAREA' ||
                                  activeElement.isContentEditable;
            
            // Skip shortcuts if input is focused (except for escape)
            if (isInputFocused && e.key !== 'Escape') {
                return;
            }
            
            // Ctrl/Cmd + T: New tab
            if ((e.ctrlKey || e.metaKey) && e.key === 't') {
                e.preventDefault();
                createNewTab();
            }
            
            // Ctrl/Cmd + W: Close tab
            if ((e.ctrlKey || e.metaKey) && e.key === 'w') {
                e.preventDefault();
                const tab = tabs.find(t => t.id === activeTabId);
                if (tab) {
                    closeTab(tab.id);
                }
            }
            
            // Ctrl/Cmd + L: Focus address bar
            if ((e.ctrlKey || e.metaKey) && e.key === 'l') {
                e.preventDefault();
                focusAddressBar();
            }
            
            // Ctrl/Cmd + R: Refresh
            if ((e.ctrlKey || e.metaKey) && e.key === 'r') {
                e.preventDefault();
                refreshPage();
            }
            
            // Alt + Left Arrow: Back
            if (e.altKey && e.key === 'ArrowLeft') {
                e.preventDefault();
                goBack();
            }
            
            // Alt + Right Arrow: Forward
            if (e.altKey && e.key === 'ArrowRight') {
                e.preventDefault();
                goForward();
            }
            
            // F11: Full screen
            if (e.key === 'F11') {
                e.preventDefault();
                toggleFullScreen();
            }
            
            // Escape: Focus address bar when not in input
            if (e.key === 'Escape' && !isInputFocused) {
                focusAddressBar();
            }
        });

        function applytheme(themeName) {

            root.style.setProperty('--secondary-bg', theme.secondaryBg);
